CREATE TABLE groups (
    PK_ID int(11) AUTO_INCREMENT PRIMARY KEY NOT NULL,
    groupName tinytext UNIQUE,
    approved BOOLEAN
);

CREATE TABLE users (
    userID int(11) UNIQUE PRIMARY KEY NOT NULL,
    email tinytext UNIQUE,
    password tinytext,
    FK_GroupID int(11)
);

CREATE TABLE reviews (
  PK_ID int(11) AUTO_INCREMENT PRIMARY KEY NOT NULL,
  rateValue int(10),
  rateJustification text,
  image BLOB,
  finalized BOOLEAN,
  Marker int(11),
  FK_MarkedUserID int(11)
);
