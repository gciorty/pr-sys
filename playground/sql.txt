CREATE TABLE images (
    PK_ID int (10) AUTO_INCREMENT PRIMARY KEY NOT NULL,
    type varchar(64),
    name varchar(64),
    alt tinytext,
    image blob
);

CREATE TABLE users (
    PK_ID int (10) AUTO_INCREMENT PRIMARY KEY NOT NULL,
    userID int(10) UNIQUE NOT NULL,
    email tinytext UNIQUE NOT NULL,
    pwdUser tinytext,
    GroupID int(10)
);

CREATE TABLE reviews (
  PK_ID int(10) AUTO_INCREMENT PRIMARY KEY NOT NULL,
  rateValue int(10),
  rateJustification text,
  image int(10),
  finalized BOOLEAN,
  FK_Marker int(10),
  FK_MarkedUserID int(10)
);
